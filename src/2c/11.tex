\begin{align*}
    \text{dim }(U + W) &= \text{dim }U + \text{dim }W - \text{dim }(U \cap W) \\
    \text{dim } \mathbf{R^8} &= 3 + 5 - \text{dim }(U \cap W) \\
    8 &= 8 - \text{dim }(U \cap W) \\
    \text{dim }(U \cap W) &= 0 \\
    U \cap W &= {0}
\intertext{Since the intersection of $U$ and $W$ is $\{0\}$ and $U + W = \mathbf{R^8}$,}
    \mathbf{R^8} &= U \oplus W
\end{align*}
